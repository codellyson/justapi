name: "Authenticated Todos"
environment:
  base: "https://jsonplaceholder.typicode.com"
defaults:
  auth: "Bearer {{ tokenVar }}"   # inherited
requests:
  - post: "{{ baseVar }}/auth/login"   # no auth yet
    bind: { token: data.access_token }
    assert: [ status = 200, data.access_token ]
  - post: "{{ baseVar }}/todos"
    bind: { todoId: data.id }
    assert: [ status = 201, data.id ]
  - get: "{{ baseVar }}/todos/{{ todoVar }}"
    assert: [ status = 200 ]
  - put: "{{ baseVar }}/todos/{{ todoVar }}"
    assert: [ status = 200, data.title~ ]
  - delete: "{{ baseVar }}/todos/{{ todoVar }}"
    assert: [ status = 200 ]
